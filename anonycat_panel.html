         from scapy.all import ARP, Ether, srp

def ip_tara(ip_araligi):
    # 1. ARP isteği oluştur (İstediğimiz IP'ye kim olduğunu soruyoruz)
    arp = ARP(pdst=ip_araligi)
    ether = Ether(dst="ff:ff:ff:ff:ff:ff") # Yayın adresi
    paket = ether / arp

    print(f"[!] Tarama başlatılıyor: {ip_araligi}...")

    # 2. Paketi gönder ve cevapları al
    cevaplar = srp(paket, timeout=3, verbose=0)[0]

    cihazlar = []
    for gonderilen, alinan in cevaplar:
        cihazlar.append({'ip': alinan.psrc, 'mac': alinan.hwsrc})

    # 3. Sonuçları listele
    print("\n--- BULUNAN CİHAZLAR ---")
    print("IP Adresi           MAC Adresi")
    print("------------------------------------------")
    for cihaz in cihazlar:
        print(f"{cihaz['ip']:18} {cihaz['mac']}")
